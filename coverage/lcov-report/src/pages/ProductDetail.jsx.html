export default function ProductDetail() {
  const { id } = useParams();
  const navigate = useNavigate();
  const { addToCart } = useCart();
 
  const product = sampleProducts.find((p) => p.id === id);
  if (!product) {
    return (
      <div style={{ padding: '2rem' }}>
        <h2>Product not found</h2>
        <Link to="/">Back to Home</Link>
      </div>
    );
  }
